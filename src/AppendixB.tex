\section{Complete list of Practical Temporal Queries}
\label{sec:appendixB/PTQs}
These queries are given in the language from \cite{borgwardt2015temporalizing} and in Java-Syntax.
\begin{enumerate}
    \item Which cars price increased some time in the past?
    \begin{itemize}
        \item $\Diamond^{-}($SELECT * FROM (SELECT * FROM $\phi$) WHERE $p1$ $<$ $p2$ [$\Circle^{-}$(SELECT $url$, $price$ AS $p1$ FROM $autos$) $\cap$ SELECT $url$, $price$ AS $p2$ FROM $autos$])
        \item Query query1 = new Even\-tually\-Past( new Filter ( ``SELECT * FROM phi WHERE p1 $<$ p2'',new Conjunction( new Strong\-Previous( new AtemporalQuery ( ``SELECT url, price AS p1 FROM autos'' ) ), new AtemporalQuery ( ``SELECT url, price AS p2 FROM autos'' ) ) ) );
    \end{itemize}

    \item Which cars price decreased by more than 20\% some time in the past?
    \begin{itemize}
        \item $\Diamond^{-}$( SELECT * FROM ( SELECT * FROM $\phi$ ) WHERE $0.8*p1$ $>$ $p2$[ $\Circle^{-}$(SELECT $url$, $price$ AS $p1$ FROM $autos$) $\cap$ SELECT $url$, $price$ AS $p2$ FROM $autos$])
        \item Query query1 = new Even\-tually\-Past ( new Filter( ``SELECT * FROM phi WHERE 0.8*p1 $>$ p2'',new Conjunction( new Strong\-Previous( new AtemporalQuery ( ``SELECT url, price AS p1 FROM autos'' ) ), new AtemporalQuery ( ``SELECT url, price AS p2 FROM autos'' ) ) ) );
    \end{itemize}

    \item For which cars did the attributes ``url'', ``price'' and ``kilometerstand'' never change?
    \begin{itemize}
        \item $\Box^{-}$($\CIRCLE^{-}$(SELECT $url$, $price$, $kilometerstand$ FROM $autos$) $\cap$ SELECT $url$, $price$, $kilometerstand$ FROM $autos$)
        \item Query query1 = new Always\-Past( new Conjunction( new Weak\-Previous( new AtemporalQuery ( ``SELECT url, price, kilometerstand FROM autos'' ) ), new AtemporalQuery ( ``SELECT url, price, kilometerstand FROM autos'' ) ) );
    \end{itemize}

    \item Which cars price decreased by more than 20\% from one time point to the next some time in the past 4 time points?
    \begin{itemize}
        \item $\Diamond^{-}_{4}$( SELECT * FROM ( SELECT * FROM $\phi$ ) WHERE $0.8*p1$ $>$ $p2$[ $\Circle^{-}$(SELECT $url$, $price$ AS $p1$ FROM $autos$) $\cap$ SELECT $url$, $price$ AS $p2$ FROM $autos$])
        \item Query query1 = new Even\-tually\-Past\-Predicate( new Filter( ``SELECT * FROM phi WHERE 0.8*p1 $>$ p2'',new Conjunction( new Strong\-Previous( new AtemporalQuery ( ``SELECT url, price AS p1 FROM autos'' ) ), new AtemporalQuery ( ``SELECT url, price AS p2 FROM autos'' ) ) ), 4 );
    \end{itemize}

    \item Which brands have an increased average price compared to the last point in time?
    \begin{itemize}
        \item SELECT $marke$ FROM (SELECT * FROM $\phi$) WHERE $p$ $<$ $q$[$\Circle^{-}$(SELECT $marke$, AVG($price$) AS $p$ FROM $autos$ GROUP BY $marke$) $\cap$ SELECT $marke$, AVG($price$) AS $q$ FROM $autos$ GROUP BY $marke$]
        \item Query query1 = new Filter ( ``SELECT marke FROM phi \linebreak WHERE p $<$ q'',new Conjunction( new Strong\-Previous( new AtemporalQuery ( ``SELECT marke, AVG(price) AS p FROM autos GROUP BY marke'' ) ), new AtemporalQuery ( ``SELECT marke, AVG(price) AS q FROM autos GROUP BY marke'' ) ) );
    \end{itemize}

    \item Which models have an increased average price compared to the last point in time?
    \begin{itemize}
        \item SELECT $modell$ FROM (SELECT * FROM $\phi$) WHERE $p$ $<$ $q$[$\Circle^{-}$(SELECT $modell$, AVG($price$) AS $p$ FROM $autos$ GROUP BY $modell$) $\cap$ SELECT $modell$, AVG($price$) AS $q$ FROM $autos$ GROUP BY $modell$]
        \item Query query1 = new Filter( ``SELECT modell FROM phi\linebreak WHERE p $<$ q'',new Conjunction( new Strong\-Previous( new AtemporalQuery ( ``SELECT modell, AVG(price) AS p FROM autos GROUP BY modell'' ) ), new AtemporalQuery ( ``SELECT modell, AVG(price) AS q FROM autos GROUP BY modell'' ) ) );
    \end{itemize}

    \item For which brands did the average price never drop from one time point to the next in the last 5 time points?
    \begin{itemize}
        \item $\Box^{-}_{5}$(SELECT $marke$ FROM ( SELECT * FROM $\phi$ ) WHERE $p$ $<=$ $q$)[$\Circle^{-}$\allowbreak(SELECT $marke$, AVG($price$) AS $p$ FROM $autos$ GROUP BY $marke$) $\cap$ SELECT $marke$, AVG($price$) AS $q$ FROM $autos$ GROUP BY $marke$]
        \item Query query1 = new Always\-Past\-Predicate(new Filter(``SELECT marke FROM phi WHERE p $<=$ q'', new Conjunction ( new Strong\-Previous ( new AtemporalQuery ( ``SELECT marke , AVG(price) AS p FROM autos GROUP BY marke'' ) ), new AtemporalQuery ( ``SELECT marke, AVG(price) AS q FROM autos GROUP BY marke'' ) ) ), 5);
    \end{itemize}

    \item For which models did the average price never drop from one time point to the next in the last 5 time points?
    \begin{itemize}
        \item $\Box^{-}_{5}$(SELECT $modell$ FROM ( SELECT * FROM $\phi$ ) WHERE $p$ $<=$ $q$)[ $\Circle^{-}$\allowbreak(SELECT $modell$, AVG($price$) AS $p$ FROM $autos$ GROUP BY $modell$) $\cap$ SELECT $modell$, AVG($price$) AS $q$ FROM $autos$ GROUP BY $modell$]
        \item Query query1 = new Always\-Past\-Predicate( new Filter(``SELECT modell FROM phi WHERE p $<=$ q'', new Conjunction( new Strong\-Previous ( new AtemporalQuery ( ``SELECT modell, AVG(price) AS p FROM autos GROUP BY modell'' ) ), new AtemporalQuery ( ``SELECT modell, AVG(price) AS q FROM autos GROUP BY modell'' ) ) ), 5);
    \end{itemize}

    \item For which brands did the amount of offers decrease?
    \begin{itemize}
        \item SELECT $marke$ FROM (SELECT * FROM $\phi$) WHERE $p$ $>$ $q$[$\Circle^{-}$(SELECT $marke$, COUNT($url$) AS $p$ FROM $autos$ GROUP BY $marke$) $\cap$ SELECT $marke$, COUNT($url$) AS $q$ FROM $autos$ GROUP BY $marke$]
        \item Query query1 =new Filter(``SELECT marke FROM phi WHERE p $>$ q'',new Conjunction( new Strong\-Previous( new AtemporalQuery ( ``SELECT marke, COUNT(url) AS p FROM autos GROUP BY marke'' ) ), new AtemporalQuery ( ``SELECT marke, COUNT(url) AS q FROM autos GROUP BY marke'' ) ) );
    \end{itemize}

    \item For which brands did the average price drop by more than 10\% some time in the past?
    \begin{itemize}
        \item $\Diamond^{-}$(SELECT $marke$ FROM (SELECT * FROM $\phi$) WHERE $0.9*p$ $>$ $q$[$\Circle^{-}$\allowbreak(SELECT $marke$, AVG($price$) AS $p$ FROM $autos$ GROUP BY $marke$) $\cap$ SELECT $marke$, AVG($price$) AS $q$ FROM $autos$ GROUP BY $marke$])
        \item Query query1 = new Even\-tually\-Past( new Filter( ``SELECT marke FROM phi WHERE 0.9*p $>$ q'',new Conjunction( new Strong\-Previous( new AtemporalQuery ( ``SELECT marke, AVG(price) AS p FROM autos GROUP BY marke'' ) ), new AtemporalQuery (``SELECT marke, AVG(price) AS q FROM autos GROUP BY marke'' ) ) ));
    \end{itemize}

    \item Which cars were always present until the time point before they were sold?
    \begin{itemize}
        \item $\Circle^{-}$($\Box^{-}$(SELECT $url$ FROM $autos$)) $\cap$ SELECT $url$ FROM $autos$ WHERE $de\-leted$
        \item Query query1 = new Conjunction( new Strong\-Previous( new Always\-Past( new AtemporalQuery ( ``SELECT url FROM autos'' ) ) ), new AtemporalQuery ( ``SELECT url FROM autos WHERE de\-leted'' ) );
    \end{itemize}

    \item Which cars cost less than 10,000 at the last point in time and cost more than 10,000 at this point in time?
    \begin{itemize}
        \item $\Circle^{-}$(SELECT $url$ FROM $autos$ WHERE $price$ $<$ 10.000) $\cap$ SELECT $url$ FROM $autos$ WHERE $price$ $>$ 10.000
        \item Query query1 = new Conjunction( new Strong\-Previous( new AtemporalQuery ( ``SELECT url FROM autos WHERE price $<$ 10.000'' ) ), new AtemporalQuery ( ``SELECT url FROM autos WHERE price $>$ 10.000'' ) );
    \end{itemize}

    \item Which cars cost more than 1.000.000 some time in the last 6 points in time?
    \begin{itemize}
        \item $\Diamond^{-}_{6}$(SELECT $url$ FROM $autos$ WHERE $price$ $>$ 1000000)
        \item Query query1 = new Even\-tually\-Past\-Predicate( new AtemporalQuery ( ``SELECT url FROM autos WHERE price $>$ 1000000'' ), 6);
    \end{itemize}

    \item Which cars had more than 100.00km at the last time point and were sold at this time point?
    \begin{itemize}
        \item $\Circle^{-}$(SELECT $url$ FROM $autos$ WHERE $kilometerstand>$\linebreak 100000) $\cap$ SELECT $url$ FROM $autos$ WHERE $de\-leted$
        \item Query query1 = new Conjunction( new Strong\-Previous( new AtemporalQuery ( ``SELECT url FROM autos WHERE kilometerstand $>$ 100000'' ) ), new AtemporalQuery ( ``SELECT url FROM autos WHERE de\-leted'' ) );
    \end{itemize}

    \item Which cars have cost more than 1.000.000 since it cost less than 1.000.000 until it was sold?
    \begin{itemize}
        \item $\Circle^{-}$(SELECT $url$ FROM $autos$ WHERE $price$ $>$ 1000000 $\mathsf{S}$ SELECT $url$ FROM $autos$ WHERE $price$ $<$ 1000000) $\cap$ SELECT $url$ FROM $autos$ WHERE $de\-leted$
        \item Query query1 = new Conjunction ( new Strong\-Previous ( new Since ( new AtemporalQuery ( ``SELECT url FROM autos\linebreak WHERE price $>$ 1000000'' ), new AtemporalQuery ( ``SELECT url FROM autos WHERE price $<$ 1000000'' ) ) ), new AtemporalQuery ( ``SELECT url FROM autos WHERE de\-leted'' ) );
    \end{itemize}
\end{enumerate}
